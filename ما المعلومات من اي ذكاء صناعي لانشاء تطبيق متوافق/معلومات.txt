أريد إنشاء تطبيق مبيعات سحابي (Cloud Sales App) متقدم واحترافي، متكامل بالكامل مع منصة MadarTech Cloud.

---

🎯 **نوع التطبيق:**
تطبيق ويب (Web App) لإدارة المبيعات مصمم خصيصاً لمندوبي المبيعات والمديرين، مع واجهة مستخدم عصرية وسهلة الاستخدام.

---

🗄️ **قاعدة البيانات (متوافقة مع D1 - SQLite):**

الجداول المطلوبة:

1. customers (العملاء):
- id, tenant_id, application_id, name, phone, email, address, city, country, customer_type (individual/company), total_purchases, total_debt, balance, status, notes, created_at, updated_at

2. products (المنتجات):
- id, tenant_id, application_id, name, barcode, sku, category, sub_category, price, cost, quantity, min_stock, max_stock, unit, weight, dimensions, image_url, description, is_active, created_at, updated_at

3. sales (المبيعات/الفواتير):
- id, tenant_id, application_id, customer_id, user_id, invoice_number, total_amount, paid_amount, remaining_amount, discount, tax, shipping_cost, sale_type (cash/credit/installment), status (pending/completed/cancelled), payment_method, notes, sale_date, created_at, updated_at

4. sale_items (تفاصيل الفواتير):
- id, tenant_id, application_id, sale_id, product_id, product_name, quantity, unit_price, total_price, discount, tax, created_at

5. payments (المدفوعات):
- id, tenant_id, application_id, sale_id, amount, payment_method, reference_number, status, payment_date, created_at

6. users (المستخدمين):
- id, tenant_id, email, password, name, role (admin/manager/sales_rep), status, created_at, updated_at

7. settings (الإعدادات):
- id, tenant_id, key, value, category, description, created_at, updated_at

typescript
// نظام المزامنة
interface SyncSystem {
    // استراتيجيات المزامنة
    strategy: 'cloud-first' | 'local-first' | 'merge';
    
    // توقيت المزامنة
    timing: 'auto' | 'manual' | 'scheduled';
    
    // التعارضات
    conflicts: Conflict[];
    
    // دوال المزامنة
    sync(): Promise<SyncResult>;
    resolveConflict(conflict: Conflict): Promise<void>;
    getStatus(): SyncStatus;
}
